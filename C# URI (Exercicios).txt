> Área (1012)

using System;
using System.Globalization;

class Programa
{
	static void Main(string[] args)
	{
		// Variaveis 1
		float A, B, C;

		// Variaveis 2
		float Triangulo, Circulo, Trapezio, Quadrado, Retangulo;

		A = float.Parse(Console.ReadLine(), CultureInfo.InvariantCulture);
		B = float.Parse(Console.ReadLine(), CultureInfo.InvariantCulture);
		C = float.Parse(Console.ReadLine(), CultureInfo.InvariantCulture);

		Triangulo = (A * C) / 2;
		Console.WriteLine($"TRIANGULO: {Triangulo:F3}", CultureInfo.InvariantCulture);

		Circulo = (float)(3.14159 * Math.Pow(C, 2));
		Console.WriteLine($"CIRCULO: {Circulo:F3}", CultureInfo.InvariantCulture);

		Trapezio = (A + B) / 2 * C;
		Console.WriteLine($"TRAPEZIO: {Trapezio:F3}", CultureInfo.InvariantCulture);

		Quadrado = (B * B);
		Console.WriteLine($"QUADRADO: {Quadrado:F3}", CultureInfo.InvariantCulture);

		Retangulo = (A * B);
		Console.WriteLine($"RETANGULO: {Retangulo:F3}", CultureInfo.InvariantCulture);
	}
}

> O Maior (1013)

using System;

class Programa
{
    static void Main(string[] args)
    {
        int a, b, c, MaiorAB, MaiorAC;

        a = int.Parse(Console.ReadLine());
        b = int.Parse(Console.ReadLine());
        c = int.Parse(Console.ReadLine());

        MaiorAB = (a + b + Math.Abs(a - b)) / 2;
        MaiorAC = (MaiorAB + c + Math.Abs(MaiorAB - c)) / 2;

        Console.WriteLine(MaiorAC + " eh o maior");

    }
}
